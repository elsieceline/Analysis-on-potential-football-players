## function for converting M and K with their respective values and for replacing the euro symbol for data manupliation
def convert_value(val):
    if isinstance(val, str):
        if val.endswith('M'):
            return float(val.replace('€', ' ').replace('M', ' '))*1000000
        elif val.endswith('K'):
            return float(val.replace('€', ' ').replace('K', ' '))*1000
        else:
            return float(val.replace('€', ''))
    else:
        return val
